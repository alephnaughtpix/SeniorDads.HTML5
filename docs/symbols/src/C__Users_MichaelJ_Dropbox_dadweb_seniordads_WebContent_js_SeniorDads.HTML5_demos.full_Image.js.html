  7 /**
  8  * @namespace SeniorDads.Image
  9  */
 10 SeniorDads.CreateNameSpace("SeniorDads.Image");
 11 
 12 /**
 13  * <h1>SeniorDads.Image</h1>
 14  * <p><strong>{@class}</strong></p>
 15  * <p><em>var myFont = new SeniorDads.Image(url, loadedCallback, autoload);</em></p>
 16  * <p>This is a resource handler which handles the loading of image resources.</p>
 17  * <p>At the moment, this quite a simple wrapper class, which only handles browser-compatible images.
 18  * however, in future, it will also handle Atari Degas format images.</p>
 19  * @class Senior Dads image resource
 20  * @param {string} url URL of the image file.
 21  * @param {function} [loadedCallback] Function to call when image is loaded.
 22  * @param {boolean} [autoload=true] Whether to autoload image.
 23  * @property {Image} element Image object.
 24  * @property {Image} imageData Image object. (Same as 'element'.)
 25  * @property {string} type The type of image. (eg "gif", "png", etc.)
 26  * @property {boolean} isBrowserImage Whether the image is a browser-compatible image.
 27  * @property {boolean} isDegasImage Whether the image is a Degas (Atari) format image.
 28  * @property {boolean} loaded Whether the resource has been loaded.
 29  * @property {number} resource_size Size of resource
 30  * @property {number} resource_loaded Resource to be loaded
 31  * @property {function} loadedCallback Function to call when image is loaded.
 32  */
 33 SeniorDads.Image = function(url, loadedCallback, autoload) {
 34     if (loadedCallback === undefined) loadedCallback = null;		// By default, there is no callback.
 35     if (autoload === undefined) autoload = true;					// By default, the image autoloaded.
 36 
 37     var type = getType(url);						// Get image type.
 38     var element = null;								// Set up container for image.
 39     var hasLoaded = false;								// Start off assuming we haven't loaded the image file.
 40     var isBrowserImage = isBrowserImage(url);		// Determine if it's a browser-compatible image.
 41     var isDegasImage = isDegasImage(url);			// Determine if it's a Degas (Atari) image.
 42     
 43     // Set up properties.
 44     this.element = function() { return element; };					// The Image object
 45     this.imageData = function () { return element; };				// The Image object (Same as 'element'.)
 46     this.type = function () { return type; };						// The type of image. (eg "gif", "png", etc.)
 47     this.isBrowserImage = function () { return isBrowserImage; };	// Whether the image is a browser-compatible image.
 48     this.isDegasImage = function () { return isDegasImage; };		// Whether the image is a Degas (Atari) format image.
 49     this.loaded = function () { 
 50     	return hasLoaded; 
 51     	};					// Whether the image has loaded
 52     this.resource_size = function () { return 1; };						 // Size of the resource
 53     this.resource_loaded =  function () { return (hasLoaded) ? 1 : 0; }; // The amount already loaded (we get this from the instance of the class we're wrapping.)
 54     // "Image loaded" callback property. This a little more complex,
 55     // as there's a "set" method which requires intervention.
 56     Object.defineProperty(this, "loadedCallback",
 57             {
 58                 get: function() { return loadedCallback; },
 59                 set: function(myLoadedCallback) {
 60                     loadedCallback = myLoadedCallback;
 61                 	// If the image is already loaded, run the callback. (Otherwise it'll never get run.)
 62                     if ((hasLoaded) && (myLoadedCallback != null)) 
 63                     	myLoadedCallback();
 64                 }
 65             });
 66 
 67     // Set up object methods
 68     /**
 69      * Load the image.
 70      * @function
 71      * @param {function} [loadingFinished] Function to call when image is loaded.
 72      */
 73     this.load = load;
 74     
 75     // If we're autoloading, (default) start the loading process.
 76     if (autoload) load(loadedCallback);
 77 
 78     /*
 79      * Load the image.
 80      * 
 81      * At the moment this handles browser-compatible images only.
 82      */
 83     function load(loadingFinished) {
 84         if (loadingFinished === undefined) loadingFinished = null;
 85         loadedCallback = loadingFinished;
 86 	    if (isBrowserImage) {				// If it's a browser image,...
 87 	        element = new Image();			// ... create new image DOM element,
 88 	        element.src = url;				// ... set it to start loading,
 89 	        element.onload = function() {	// When it's loaded,...
 90 	        	hasLoaded = true;			// ... set the loaded flag.
 91 	            if (loadedCallback != null) loadedCallback();	// If there's a callback, run that.
 92 	        };
 93 	    }
 94     }
 95 
 96     /*
 97      * Test if it's a browser-compatible image.
 98      */
 99     function isBrowserImage(imageUrl) {
100         return (imageUrl.match(/\.(jpeg|jpg|gif|png)$/) != null);
101     }
102 
103     /*
104      *  Test if it's a Degas image.
105      *  
106      *  Degas Elite (as well as it's predecessor Degas) was a very popular 
107      *  art program on the Atari ST. It was developed by Tom Hudson in 1986, 
108      *  and published by Batteries Included software. 
109      *  
110      *  The uncompressed version of the picture format was literally a full ST 
111      *  colour palette setup as it would be stored in the ST palette registers, 
112      *  followed by the picture as it would be stored in ST screen memory. 
113      *  As a result, the picture format (Which other arts programs on the ST 
114      *  such as Deluxe Paint ST could load and save.) was very popular with coders, 
115      *  and got used a lot for graphics in Senior Dads demos!
116      *  
117      */
118     function isDegasImage(imageUrl) {
119         return (imageUrl.match(/\.(pc?|pi?)$/) != null);
120     }
121     
122     /*
123      * Get the type of image.
124      * 
125      * This returns the file extension of the image URL.
126      */
127     function getType( imageFilename ) {
128     	var startCheck = imageFilename.lastIndexOf(".");	// Find where the file extension is.
129     	var type = null;									// Return value if we don't find a file extension.
130     	if (startCheck > -1)								// if there's a file extension, extract it.
131     		type = imageFilename.substring(startCheck+1, imageFilename.length - startCheck);
132     	return type;
133     }
134 };
