  4 SeniorDads.ScreenHandler = function (element, name, width, height) {
  5     var screenWrapper = element.appendChild(document.createElement('div'));
  6     var orderPointer = 1;
  7     var screenList = new Object;
  8     var imageData;
  9     var waitVblAnim;
 10     screenWrapper.id = name;
 11     screenWrapper.style.width = width + "px";
 12     screenWrapper.style.height = height + "px";
 13     screenWrapper.style.position = "relative";
 14 
 15     this.add = add;
 16     this.WaitVbl = waitVbl;
 17     this.FlushWaitVbl = flushWaitVbl;
 18 
 19     function add(myName, scale, image) {
 20         screenList[name] = new SeniorDads.ScreenHandler.Screen(screenWrapper, myName, width, height, scale, orderPointer++, image);
 21         return screenList[name];
 22     }
 23 
 24     function waitVbl(noOfVBLs, callback) {
 25         waitVblAnim = setTimeout(function () {
 26         	window.requestAnimFrame(callback);
 27         }, noOfVBLs * 20);
 28     }
 29 
 30     function flushWaitVbl() {
 31         clearTimeout(waitVblAnim);
 32     }

 34     function goFullScreen() {
 35 		var elem = element;
 36 		if (elem.requestFullscreen) {
 37 			elem.requestFullscreen();
 38 		} else if (elem.mozRequestFullScreen) {
 39 			elem.mozRequestFullScreen();
 40 		} else if (elem.webkitRequestFullScreen) {
 41 			elem.webkitRequestFullScreen(Element.ALLOW_KEYBOARD_INPUT);
 42 		}
 43     }

 45     SeniorDads.ScreenHandler.Screen = function (element, name, width, height, scale, zIndex, image) {
 46         if (image === undefined) image = null;
 47         var loaded = false;
 48         var backgroundColour;
 49         var canvas = element.appendChild(document.createElement('canvas'));
 50         canvas.id = name;
 51         var scaledWidth = width * scale;
 52         var scaledHeight = height * scale;
 53         canvas.width = scaledWidth;
 54         canvas.height = scaledHeight;
 55         canvas.style.position = "absolute";
 56         canvas.style.top = "0px";
 57         canvas.style.left = "0px";
 58         canvas.style.zIndex = zIndex;
 59         var context = canvas.getContext('2d');
 60         context.setTransform(1, 0, 0, 1, 0, 0); // Reset scaling
 61         context.scale(scale, scale);            // Set scaling 
 62         context.imageSmoothingEnabled = false;  // Switch off interpolation
 63         hide();
 64 
 65         if (image != null)
 66             if (image.loaded()) {
 67                 drawImage();
 68             } else {
 69                 image.loadedCallback = drawImage;
 70             }
 71         else
 72             loaded = true;
 73 
 74         this.loaded = function () { return loaded; };
 75         this.resource_size = function () { return 1; };					// Size of the resource
 76         this.resource_loaded =  function () { return 0; }; 				// The amount already loaded (we get this from the instance of the class we're wrapping.)
 77         this.canvas = function () { return canvas; };
 78         this.context = function () { return context; };
 79         this.name = function () { return name; };
 80         this.zIndex = function () { return zIndex; };
 81         this.width = function () { return width; };
 82         this.height = function () { return height; };
 83         this.scale = function () { return scale; };
 84 
 85         this.Show = show;
 86         this.Hide = hide;
 87         this.Clear = clear;
 88 
 89         function drawImage() {
 90             if (!(image == null)) {
 91                 imageData = image.imageData();
 92                 context.drawImage(imageData, 0, 0);
 93                 loaded = true;
 94             }
 95         }
 96 
 97         function clear(myColour) {
 98             if (myColour === undefined)
 99                 myColour = backgroundColour;
100             else
101                 backgroundColour = myColour;
102             if (myColour != null) {
103                 context.fillStyle = myColour;
104                 context.fillRect(0, 0, width, height);
105             }
106         }
107 
108         function show() {
109             canvas.style.visibility = 'visible';
110         }
111 
112         function hide() {
113             canvas.style.visibility = 'hidden';
114         }
115     };

125     SeniorDads.ScreenHandler.Codef = function( width, height, name, zIndex, x, y )  {
126     	var codefCanvas;
127     	if (x === undefined) x = 0;
128     	if (y === undefined) y = 0;
129         if (name === undefined) 
130         	codefCanvas = new canvas(width,height);
131         else {
132         	codefCanvas = new canvas(width,height,name);
133 	        codefCanvas.canvas.style.position = "absolute";
134 	        codefCanvas.canvas.style.left = x + "px";
135 	        codefCanvas.canvas.style.top = y +"px";
136 	        if (!(zIndex === undefined))
137 	        	codefCanvas.canvas.style.zIndex = zIndex;
138 
139         }
140         codefCanvas.canvas.style.visiblity = 'hidden';
141         Object.defineProperties(codefCanvas, {
142             show: {
143                 value: function () {
144                 	SeniorDads.ScreenHandler.show(this.canvas);
145                 }
146             },
147             hide: {
148                 value: function () {
149                 	SeniorDads.ScreenHandler.hide(this.canvas);
150                 }
151             },
152             x: {
153             	set: function(x) {
154             		this.canvas.style.left = x + "px";
155             	}
156             },
157             y: {
158             	set: function(y) {
159             		this.canvas.style.top = y + "px";
160             	}
161             }
162         } );
163         return codefCanvas;
164     };
