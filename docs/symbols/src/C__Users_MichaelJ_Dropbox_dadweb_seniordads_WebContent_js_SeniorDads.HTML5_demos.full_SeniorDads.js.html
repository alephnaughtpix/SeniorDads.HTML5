 12 	return  window.requestAnimationFrame       || 
 13 		window.webkitRequestAnimationFrame ||

 36 // create a general purpose namespace method
 37 // this will allow us to create namespace a bit easier
 38 SeniorDads.CreateNameSpace = function (namespace) {
 39     var nsparts = namespace.split(".");
 40     var parent = SeniorDads;
 41  
 42     // we want to be able to include or exclude the root namespace 
 43     // So we strip it if it's in the namespace
 44     if (nsparts[0] === "SeniorDads") {
 45         nsparts = nsparts.slice(1);
 46     }
 47  
 48     // loop through the parts and create 
 49     // a nested namespace if necessary
 50     for (var i = 0; i < nsparts.length; i++) {
 51         var partname = nsparts[i];
 52         // check if the current parent already has 
 53         // the namespace declared, if not create it
 54         if (typeof parent[partname] === "undefined") {
 55             parent[partname] = {};
 56         }
 57         // get a reference to the deepest element 
 58         // in the hierarchy so far
 59         parent = parent[partname];
 60     }
 61     // the parent is now completely constructed 
 62     // with empty namespaces and can be used.
 63     return parent;
 64 };

 72 /**
 73  * <h1>Wobbler</h1>
 74  * <p><strong>Create a "wobble" effect.</strong></p>
 75  * <p><em>var myWobbler = new SeniorDads.Wobbler(hWobbleParameters, vWobbleParameters, zWobbleParameters);</em></p>
 76  * @constructor
 77  * @class Create a "wobble" effect.
 78  * @param {Object[]} hWobbleParameters Horizontal wobble parameters.
 79  * @param {Object[]} [vWobbleParameters] vertical wobble parameters.
 80  * @param {Object[]} [zWobbleParameters] Z-index wobble parameters.
 81  * @property {number} h Get next horizontal wobble differential.
 82  * @property {number} v Get next vertical wobble differential.
 83  * @property {number} z Get next Z-index wobble differential.
 84  * @example
 85  *  //Place an image in the center of a 640x400 area and wobble it.
 86  *  
 87  *  // First create the area and load the image
 88  *  var myPlaceholder = document.getElementById('placeholder'); // <- Assuming we have a div with ID of 'placeholder' on the web page.
 89  *  myPlaceholder.style.width = "640px";
 90  *  myPlaceholder.style.height = "400px";
 91  *  myPlaceholder.style.position = "relative";
 92  *  var myImage = new Image();
 93  * 	myImage = "myImage_url.png";
 94  * 	myImage.style.position = "absolute";
 95  *  myPlaceholder.appendChild(myImage);
 96  *  
 97  *  var myWobbler;
 98  *  
 99  *  // When the image is loaded, create the wobbler
100  *  myImage.onload = function() {
101  *  
102  *  	myWobbler = new SeniorDads.Wobbler(
103  *  		[ 
104  *  			{value: 0, amp: 3, inc: 0.30},
105  *  			{value: 0.5, amp: 3, inc: 0.40}
106  *  		],
107  *  		[
108  *  			{value: 0.5, amp: 3, inc: 0.20},
109  *  			{value: 0, amp: 2, inc: 0.10}
110  *  		]);
111  *  	// Start the animation loop.
112  *  	animationLoop();
113  *  }
114  *  
115  *  // Animation loop- wobbles the image relative to the centre of the area.
116  *  function animationLoop() {
117  *  	var x = 320 + myWobbler.h();
118  *  	var y = 200 + myWobbler.v();
119  *  	myImage.style.left = x + "px";
120  *  	myImage.style.top = y + "px";
121  *  	requestAnimFrame( animationLoop );
122  *  }
123  */
124 SeniorDads.Wobbler = function (hWobbleParameters, vWobbleParameters, zWobbleParameters) {
125 	if (hWobbleParameters === undefined) hWobbleParameters = null;
126 	if (vWobbleParameters === undefined) vWobbleParameters = null;	// Default for optional parameter
127 	if (zWobbleParameters === undefined) zWobbleParameters = null;	// Default for optional parameter
128 	
129 	// Set properties
130 	this.h = function() { return wobble(hWobbleParameters); };		// get next horizontal wobble differential
131 	// Get next vertical wobble differential
132 	if ( ( hWobbleParameters != null) && (vWobbleParameters == null) )
133 		this.v = this.h;											// If only "horizontal" wobble supplied, vertical differential = horizontal
134 	else
135 	// Get next z-index wobble differential
136 	this.v = function() { return wobble(vWobbleParameters); };
137 	if ( ( hWobbleParameters != null) && (zWobbleParameters == null) )
138 		this.z = this.h;											// If only "horizontal" wobble supplied, z-index differential = horizontal
139 	else
140 		this.z = function() { return wobble(zWobbleParameters); };
141 		
142 	/* Main wobbling calculator.
143 	 * 
144 	 * This combines the sines waves and their positions as defined in the 
145 	 * wobble parameters to arrive at a combined wobble value.
146 	 * 
147 	 */
148 	function wobble(wobbleParameters) {
149 		var wobbleVal = null;			// Default value for invalid wobble.
150 		if (wobbleParameters != null)	// only calculate if we have wobble parameters.
151 			wobbleVal = 0;				// Reset wobble value.
152 			for( var i = 0; i < wobbleParameters.length; i++) { 							// For each wobble param...
153 				wobbleVal += Math.sin(wobbleParameters[i].value) * wobbleParameters[i].amp; // Add current sine wave position of wobble element
154 				wobbleParameters[i].value += wobbleParameters[i].inc;						// Move onto next sine wave position.
155 			}
156 		// Combined wobble value has now been calculated.
157 		return wobbleVal; 
158 	}
159 	
160 };

162 /**
163  * <h1>Bomb</h1>
164  * <p><strong>Display the Atari TOS "bomb" pattern for exceptions in Atari 16/24bit machines.</strong></p>
165  * <p><em>var myBomb = new SeniorDads.Bomb();</em></p>
166  * @class Display the Atari TOS "bomb" pattern.
167  * @constructor
168  * @example
169  *  // First create the canvas to bomb.
170  *  var myPlaceholder = document.getElementById('placeholder'); // <- Assuming we have a div with ID of 'placeholder' on the web page.
171  *  var myCanvas = myPlaceholder.appendChild(document.createElement('canvas'));
172  *  
173  *  // Now bomb it!
174  *  var myBomb = new SeniorDads.Bomb();
175  *  myBomb.bombCanvas(myCanvas, 4);
176  */
177 SeniorDads.Bomb = function () {
178     // Atari TOS bomb bitmap pattern- copied from old demo source code, itself copied from ROM listing in ST Internals!
179     var bombBits = new Uint16Array([
180         1536,  // %0000011000000000
181         10496, // %0010100100000000
182         128,   // %0000000010000000
183         18496, // %0100100001000000
184         4592,  // %0001000111110000
185         496,   // %0000000111110000
186         2044,  // %0000011111111100
187         4094,  // %0000111111111110
188         4094,  // %0000111111111110
189         8191,  // %0001111111111111
190         8175,  // %0001111111101111
191         4078,  // %0000111111101110
192         4062,  // %0000111111011110
193         2044,  // %0000011111111100
194         1016,  // %0000001111111000
195         224    // %0000000011100000
196     ]);
197     // Set up a canvas for the canvas friendly version of the bomb image.
198     var truColBombCanvas = document.createElement('canvas');
199     truColBombCanvas.id = "SeniorDads.Bomb.Store";
200     truColBombCanvas.width = 16;					// Bomb is 16x16 pixels.
201     truColBombCanvas.height = 16;
202     truColBombCanvas.style.visibility = 'hidden';
203 
204     convertBombToTruCol();			// Convert the bitmap to the canvas friendly version of the bomb image.
205 
206     /**
207      * <h3>bombCanvas</h3>
208      * 
209      * <p><strong>Display bombs.</strong></p>
210      * <p>The bombs are drawn in black on white in the middle left of the screen.</p>
211      * @function
212      * @param {SeniorDads.ScreenHandler.Screen} screen Screen to draw to.
213      * @param {number} bombCount Number of bombs to draw
214      */
215     this.bombCanvas = bombCanvas;	// Setup method for creating bombs
216 
217     /*
218      * Convert bomb bitmap to canvas colours.
219      * 
220      * For each pixel we set the foreground/background colour accordingly. 
221      * ie black on white, as used in the Atari Falcon in trucolor mode.
222      *  
223      */
224     function convertBombToTruCol() {
225         var foreground = [0, 0, 0, 0xff];					// Foreground is black.
226         var background = [0xff, 0xff, 0xff, 0xff];			// Background is white.
227         var context = truColBombCanvas.getContext('2d');
228         var canvasData = context.createImageData(16, 16);	// Create the 16x16 image data.
229         var colour;
230         var canvasDataPosition = 0;							// Start at the top left of the image data.
231         for (var i = 0; i < 16; i++) {						// There's 16 words in the bomb,
232             var currentBombLine = bombBits[i];				// ... each of which equals a single row of the bomb pixels,
233             for (var j = (16 - 1); j > -1; --j) {			// ... which is 16 pixels in length
234             	// Bit-test current pixel in word to determine if there's a pixel present.
235             	// If there is use the foreground colour, otherwise the background colour.
236                 colour = ((currentBombLine & (1 << j)) > 0) ? foreground : background;
237             	// Now plot the pixel in the relevant colour.
238                 canvasData.data[canvasDataPosition++] = colour[0];
239                 canvasData.data[canvasDataPosition++] = colour[1];
240                 canvasData.data[canvasDataPosition++] = colour[2];
241                 canvasData.data[canvasDataPosition++] = colour[3];
242             }
243         }
244         // Now the bomb is drawn, copy it to the bomb canvas.
245         context.putImageData(canvasData, 0, 0);
246     }
247     
248     /*
249      * bombCanvas
250      * # screen - SeniorDads.ScreenHandler.Screen to draw to.
251      * # bombCount - number of bombs to draw
252      */
253     function bombCanvas( screen, bombCount ) {
254     	var y = (screen.height() - 16) / 2;			// Go to middle left of screen
255     	var x = 0;
256     	var context = screen.context();				// Get canvas context of screen
257     	for (var i=0; i < bombCount; i++) {				// For the defined number of bombs...
258     		context.drawImage(truColBombCanvas, x, y);	// Copy the bomb canvas to screen.
259     		x += 16;									// Move to position of next bomb.
260     	} 
261     };
262 };

271 /**
272  * <h1>SeniorDads.Demos.Demo</h1>
273  * <p><strong>Senior Dads demo framework.</strong></p>
274  * <p><em>var myDemo = new SeniorDads.Demos.Demo(preLoader, loader, init, abort);</em></p>
275  * <p>This class allows you to add individual screens (See {@link SeniorDads.Demos.Demo.DemoPart}), and to handle
276  * general loading of resources, and initialisation before starting the demo. There is also a <i>preloader</i>, which can be
277  * used if you need to set something up before loading resources. (eg a "Now loading demo" type screen.)</p>
278  * <p>Once the demo is started, the class can handle moving through the list of screens, or stopping the demo altogether.</p>
279  * <p>Both the preloader and loaders can be supplied with callback functions to be executed when they're done.<p>
280  * @constructor 
281  * @class Senior Dads demo framework
282  * @param {function} preLoader Function to call <i>before</i> loading resources
283  * @param {function} loader Function to call to <i>start</i> loading resources
284  * @param {function} init Function to call after loading resources but <i>before</i> running the demo.
285  * @param {function} abort Function to call if the demo needs to make a speedy exit.
286  */
287 SeniorDads.Demos.Demo = function(preLoader, loader, init, abort) {
288 	var currentDemoPart;			// Currently running SeniorDads.Demos.Demo.DemoPart 
289 	var demoPartList = [];			// Array of SeniorDads.Demos.Demo.DemoPart containing the demo screens.
290 	var currentDemoPartCounter = 0; // Current position of demo
291     
292 	/**
293 	 * Preload a demo, without initalising or starting.
294 	 * @function
295 	 * @param {function} [preLoaderDone] Function to call when preloader is finished.
296 	 */
297     this.preload = preLoad;
298     
299 	/**
300 	 * Load a demo, without initalising or starting.
301 	 * @function
302 	 * @param {function} [loaderDone] Function to call when loader is finished.
303 	 */
304     this.load = load;
305 	/**
306 	 * Initialise a demo, without starting.
307 	 * @function
308 	 */
309     this.init = initialise; 
310 	/**
311 	 * Abort a demo.
312 	 * @function
313 	 */
314     this.abort = abortDemo;
315 	/**
316 	 * Add a screen to a demo. This requires creation of a {@link SeniorDads.Demos.Demo.DemoPart}.
317 	 * @example
318 	 *  // Example stub for a simple one screen demo.
319 	 *  
320 	 *  // Create the demo.
321 	 *  var myDemo = new SeniorDads.Demos.Demo(preLoader, loader, init, abort);
322 	 *  // Add a screen to it.
323 	 *  myDemo.add( new SeniorDads.Demos.Demo.DemoPart( part1init, part1Start, part1abort );
324 	 *  // Load and start the demo.
325 	 *  myDemo.loadAndStart();
326 	 *  
327 	 *  // [...]
328 	 *  
329 	 *  // PRELOADER:
330 	 *  // This is run first by the demo object, with a callback to the 
331 	 *  // loader function in the parameter "preLoaderDone".
332 	 *  function preLoader( preLoaderDone ) {
333 	 *  	// [... Pre loading code ...]
334 	 *  	preLoaderDone();	// <- This will call the function "loader" below.
335 	 *  }
336 	 *  
337 	 *  // LOADER:
338 	 *  // This is run second as a result of the preloader callback. It contains
339 	 *  // a callback to the initalise and start function in the parameter "loaderDone".
340 	 *  function loader( loaderDone ) {
341 	 *  	// [... Loading code ...]
342 	 *  	loaderDone() 	// <- This will call the initialise and start functions in the object.
343 	 *  }
344 	 *  
345 	 *  // INITIALISE:
346 	 *  // This is run third by the demo object's initialise and start functions.
347 	 *  //
348 	 *  // After this is run, the first demo screen is run (ie function 'part1Start'.)
349 	 *  function init() {
350 	 *  	// [... Initalisation code ...]
351 	 *  }
352 	 *  
353 	 *  // ABORT:
354 	 *  // This is called if "myDemo.abort();" is used.
355 	 *  function abort() {
356 	 *  	// [... Demo abort code ...]
357 	 *  }
358 	 *  
359 	 *  // DEMO SCREEN INITIALISE:
360 	 *  // This is run when the demo screen is first added to the demo. 
361 	 *  // (ie before the start of the demo, and as a result of the command 
362 	 *  // "myDemo.add( ... )".)
363 	 *  function part1init() {
364 	 *  	// [... Screen initialisation code ...]
365 	 *  }
366 	 *  
367 	 *  // DEMO SCREEN RUN:
368 	 *  // This is run when the demo object runs the first screen.
369 	 *  function part1Start() {
370 	 *  	// [... Screen main code ...]
371 	 *  }
372 	 *  
373 	 *  // DEMO SCREEN ABORT:
374 	 *  // This is called if "myDemo.abort();" is used.
375 	 *  function part1abort() {
376 	 *  	// [... Screen abort code ...]
377 	 *  }
378 	 * @function
379 	 * @param {SeniorDads.Demos.Demo.DemoPart} currentDemo Demo screen.
380 	 */
381     this.add = addDemoPart;
382 	/**
383 	 * Start demo.
384 	 * @function
385 	 */
386     this.start = start;
387 	/**
388 	 * Go to next demo screen.
389 	 * @function
390 	 */
391     this.next = nextDemoPart;
392 	/**
393 	 * Load, initialise and start demo. This will call, in the following order:
394 	 * <ul>
395 	 * <li>the preloader,</li>
396 	 * <li>the loader,</li>
397 	 * <li>the initialisation,</li>
398 	 * <li>and finally, the demo, from the first screen onwards.</li>
399 	 * @function
400 	 */
401     this.loadAndStart = loadAndStart;
402     
403     /*
404      * Add demo part.
405      * 
406      * This pushes the demo screen in "currentDemo" onto the array of SeniorDads.Demos.Demo.DemoPart
407      */
408     function addDemoPart(currentDemo) {
409     	demoPartList.push(currentDemo);
410     }
411     
412     /*
413      * Add demo parts. (Experimental.)
414      * 
415      * This accepts an array of SeniorDads.Demos.Demo.DemoPart in "currentDemo" into the demo screen array.
416      */
417     function addDemoParts(newDemoList) {
418     	demoPartList = newDemoList.clone();
419     }
420 
421     /*
422      * Load demo, initialise and start. 
423      * 
424      * This results in the following being called in order:
425      * # preLoader (if it exists)
426      * # loader (if it exists)
427      * # init (if it exists)
428      * # start (if it exists)
429      */
430     function loadAndStart() { 
431     	preLoad( function() { load( initAndStart ); } ); 
432     }
433     
434     /*
435      * Run demo preloader, and then run the callback in "preLoaderDone" when finished.
436      */
437     function preLoad( preLoaderDone ) {
438         if (preLoaderDone === undefined) preLoaderDone = null;	// "preLoaderDone" is optional.
439     	if (preLoader != null)  		// Only run if we have an preloader function
440     		preLoader( preLoaderDone );
441     	else  							// If there's no preloader, try running the callback
442     		if (preLoaderDone != null)
443     			preLoaderDone();
444     }
445     
446     /*
447      * Run demo loader, and then run the callback in "loaderDone" when finished.
448      */
449     function load( loaderDone ) {
450         if (loaderDone === undefined) loaderDone = null;	// "loaderDone" is optional.
451     	if (loader != null)  			// Only run if we have an loader function
452     		loader( loaderDone );
453     	else  							// If there's no loader, try running the callback
454     		if (loaderDone != null)
455     			loaderDone();
456     }
457 
458     /*
459      * Initialise and start demo. 
460      * 
461      * This results in the following being called in order:
462      * # init (if it exists)
463      * # start (if it exists)
464      */
465     function initAndStart() {
466     	initialise(); 
467 		start();
468     }
469 
470     /*
471      * Initialise demo without starting. 
472      */
473     function initialise() {
474     	if ( init != null)		// Only run if we have an initialise function
475     		init(); 
476     }
477     
478     /*
479      * Start demo.
480      * 
481      * This resets the demo array position to the start, and pops off the 
482      * next demo screen. (ie the first one.)
483      */
484     function start() {
485     	currentDemoPartCounter = 0;			// Start of demo
486     	nextDemoPart();						// Start first demo screen
487     }
488     
489     /*
490      * Get next demo screen.
491      * 
492      * This pops off the next screen in the array. If we're out of screens,
493      * it evokes the demo abort.
494      */
495     function nextDemoPart() {
496     	if (currentDemoPartCounter < demoPartList.length ) {			// If We've still got screens,...
497 	    	currentDemoPart = demoPartList[currentDemoPartCounter++];	// ... get the next screen,...
498 	    	currentDemoPart.start();									// ... and start it.
499     	}
500     	else {
501     		abortDemo();
502     	}
503     }
504     
505     /*
506      * Abort demo.
507      * 
508      * If we're got any screens currently running, abort that as well.
509      */
510     function abortDemo() {
511     	if ( currentDemoPart != null )			// If we're got any screens currently running...
512     		currentDemoPart.abort();			// .. abort that as first.
513     	if ( abort != null)						// Only run if we have an abort function
514     		abort();
515     }
516     
517     /**
518     * <h1>SeniorDads.Demos.Demo.DemoPart</h1>
519     * <p><strong>Senior Dads demo screen.</strong></p>
520     * <p><em>var myDemoPart = new SeniorDads.Demos.Demo.DemoPart(init, start, abort);</em></p>
521     * @constructor 
522     * @class Senior Dads demo screen.
523 	* @param {function} init Function to call before starting.
524 	* @param {function} start Function to call to start screen.
525 	* @param {function} abort Function to call if the screen needs to make a speedy exit.
526     */
527     SeniorDads.Demos.Demo.DemoPart = function(init, start, abort) {
528     	/**
529     	 * Call function before starting.
530     	 * @function
531     	 */
532         this.init = function () { 
533         	if ( init != null)				// Only call if we have a function.
534         		init(); 
535         };
536     	/**
537     	 * Call function to start screen.
538     	 * @function
539     	 */
540         this.start = function () { 
541         	if ( start != null)				// Only call if we have a function.
542         		start(); 
543         };
544     	/**
545     	 * Call function if the screen needs to make a speedy exit.
546     	 * @function
547     	 */
548         this.abort = function () { 
549         	if ( abort != null)				// Only call if we have a function.
550         		abort(); 
551         };
552         
553         this.init();
554     };
555 };
