  3 // of his player for the Javascript soundtracker SoundBox (previously known as

  8 /* Constructor for SoundBox class.
  9  * 
 10  */
 11 SeniorDads.Music.SoundBox = function( song_json, loadedCallback, checkInterval ) {
 12     if (loadedCallback === undefined) loadedCallback = null;	// If there's no call back 
 13     if (checkInterval === undefined) checkInterval = 500;		// Default for optional value
 14     var loaded = false;				// Start off assuming we haven't loaded the file.
 15     var resource_loaded = 0;		//
 16     
 17     var audio = document.createElement("audio");				// HTML5 Audio element for playing the song
 18 
 19 	var musicPlayer = new CPlayer();
 20 	musicPlayer.init(song_json);
 21 
 22     // PROPERTIES
 23     this.loaded = function () { return loaded; };					 // Whether the file has loaded
 24     this.resource_size = function () { return 8; };					 // Size of the resource
 25     this.resource_loaded =  function () { return resource_loaded; }; // The amount already loaded (we get this from the instance of the class we're wrapping.)
 26     this.loop = audio.loop;
 27     
 28     // METHODS
 29     this.play = play;
 30     this.getData = musicPlayer.getData;
 31     this.currentTime = audio.currentTime;
 32     this.duration = audio.duration;
 33     this.waitUntilEnd = waitUntilEnd;
 34 
 35     var loaderCheckInterval = setInterval(checkSongGeneration, checkInterval);
 36 
 37 
 38     /* Checking the song generation process. 
 39      * 
 40      * This is called every so often (usually every 500ms) until generation is completed.
 41      */
 42     function checkSongGeneration() {
 43     	if (loaded) {	 // If it's complete
 44             clearInterval(loaderCheckInterval);	 // ... Stop the process
 45             return;
 46     	}
 47     	var completed = (musicPlayer.generate() >= 1);
 48         if (completed) {						 // If it's complete
 49             clearInterval(loaderCheckInterval);	 // ... Stop the process
 50             var wave = musicPlayer.createWave(); // ... Create the WAV file of the song
 51             audio.src = URL.createObjectURL(new Blob([wave], {type: "audio/wav"})); // ... and load it into the Audio element
 52             audio.loop = false;					 // ... by default, switch off looping
 53             loaded = true;						 // ... set the loaded flag 
 54             if (loadedCallback != null)			 // ... and if there's a callback
 55                 loadedCallback();				 // ... call it.
 56         }
 57         else 
 58         	resource_loaded++;
 59     }
 60     
 61     function play() {
 62     	if (loaded)
 63     		audio.play();
 64     }
 65     
 66     function waitUntilEnd(callback) {
 67     	if (callback === undefined) callback = null;
 68     	if (!(callback == null))
 69     		if (loaded)
 70     			audio.onended = callback;
 71     }
 72 };

101 var CPlayer = function() {
102 
103     //--------------------------------------------------------------------------
104     // Private methods
105     //--------------------------------------------------------------------------
106 
107     // Oscillators
108     var osc_sin = function (value) {
109         return Math.sin(value * 6.283184);
110     };
111 
112     var osc_saw = function (value) {
113         return 2 * (value % 1) - 1;
114     };
115 
116     var osc_square = function (value) {
117         return (value % 1) < 0.5 ? 1 : -1;
118     };
119 
120     var osc_tri = function (value) {
121         var v2 = (value % 1) * 4;
122         if(v2 < 2) return v2 - 1;
123         return 3 - v2;
124     };
125 
126     var getnotefreq = function (n) {
127         // 174.61.. / 44100 = 0.003959503758 (F3)
128         return 0.003959503758 * Math.pow(2, (n - 128) / 12);
129     };
130 
131     var createNote = function (instr, n) {
132         var osc1 = mOscillators[instr.i[0]],
133             o1vol = instr.i[1],
134             o1xenv = instr.i[3],
135             osc2 = mOscillators[instr.i[4]],
136             o2vol = instr.i[5],
137             o2xenv = instr.i[8],
138             noiseVol = instr.i[9],
139             attack = instr.i[10] * instr.i[10] * 4,
140             sustain = instr.i[11] * instr.i[11] * 4,
141             release = instr.i[12] * instr.i[12] * 4,
142             releaseInv = 1 / release;
143 
144         var noteBuf = new Int32Array(attack + sustain + release);
145 
146         // Calculate note frequencies for the oscillators
147         var o1t = getnotefreq(n + instr.i[2] - 128);
148         var o2t = getnotefreq(n + instr.i[6] - 128) * (1 + 0.0008 * instr.i[7]);
149 
150         // Re-trig oscillators
151         var c1 = 0, c2 = 0;
152 
153         // Local variables.
154         var j, e, t, rsample;
155 
156         // Generate one note (attack + sustain + release)
157         for (j = 0; j < attack + sustain + release; j++) {
158             // Envelope
159             e = 1;
160             if (j < attack) {
161                 e = j / attack;
162             } else if (j >= attack + sustain) {
163                 e -= (j - attack - sustain) * releaseInv;
164             }
165 
166             // Oscillator 1
167             t = o1t;
168             if (o1xenv) {
169                 t *= e * e;
170             }
171             c1 += t;
172             rsample = osc1(c1) * o1vol;
173 
174             // Oscillator 2
175             t = o2t;
176             if (o2xenv) {
177                 t *= e * e;
178             }
179             c2 += t;
180             rsample += osc2(c2) * o2vol;
181 
182             // Noise oscillator
183             if (noiseVol) {
184                 rsample += (2 * Math.random() - 1) * noiseVol;
185             }
186 
187             // Add to (mono) channel buffer
188             noteBuf[j] = (80 * rsample * e) | 0;
189         }
190 
191         return noteBuf;
192     };
193 
194 
195     //--------------------------------------------------------------------------
196     // Private members
197     //--------------------------------------------------------------------------
198 
199     // Array of oscillator functions
200     var mOscillators = [
201         osc_sin,
202         osc_square,
203         osc_saw,
204         osc_tri
205     ];
206 
207     // Private variables set up by init()
208     var mSong, mLastRow, mCurrentCol, mNumWords, mMixBuf;
209 
210 
211     //--------------------------------------------------------------------------
212     // Initialization
213     //--------------------------------------------------------------------------
214 
215     this.init = function (song) {
216         // Define the song
217         mSong = song;
218 
219         // Init iteration state variables
220         mLastRow = song.endPattern - 2;
221         mCurrentCol = 0;
222 
223         // Prepare song info
224         mNumWords =  song.rowLen * song.patternLen * (mLastRow + 1) * 2;
225 
226         // Create work buffer (initially cleared)
227         mMixBuf = new Int32Array(mNumWords);
228     };
229 
230 
231     //--------------------------------------------------------------------------
232     // Public methods
233     //--------------------------------------------------------------------------
234 
235     // Generate audio data for a single track
236     this.generate = function () {
237         // Local variables
238         var i, j, b, p, row, col, n, cp,
239             k, t, lfor, e, x, rsample, rowStartSample, f, da;
240 
241         // Put performance critical items in local variables
242         var chnBuf = new Int32Array(mNumWords),
243             instr = mSong.songData[mCurrentCol],
244             rowLen = mSong.rowLen,
245             patternLen = mSong.patternLen;
246 
247         // Clear effect state
248         var low = 0, band = 0, high;
249         var lsample, filterActive = false;
250 
251         // Clear note cache.
252         var noteCache = [];
253 
254          // Patterns
255          for (p = 0; p <= mLastRow; ++p) {
256             cp = instr.p[p];
257 
258             // Pattern rows
259             for (row = 0; row < patternLen; ++row) {
260                 // Execute effect command.
261                 var cmdNo = cp ? instr.c[cp - 1].f[row] : 0;
262                 if (cmdNo) {
263                     instr.i[cmdNo - 1] = instr.c[cp - 1].f[row + patternLen] || 0;
264 
265                     // Clear the note cache since the instrument has changed.
266                     if (cmdNo < 14) {
267                         noteCache = [];
268                     }
269                 }
270 
271                 // Put performance critical instrument properties in local variables
272                 var oscLFO = mOscillators[instr.i[13]],
273                     lfoAmt = instr.i[14] / 512,
274                     lfoFreq = Math.pow(2, instr.i[15] - 9) / rowLen,
275                     fxLFO = instr.i[16],
276                     fxFilter = instr.i[17],
277                     fxFreq = instr.i[18] * 43.23529 * 3.141592 / 44100,
278                     q = 1 - instr.i[19] / 255,
279                     dist = instr.i[20] * 1e-5,
280                     drive = instr.i[21] / 32,
281                     panAmt = instr.i[22] / 512,
282                     panFreq = 6.283184 * Math.pow(2, instr.i[23] - 9) / rowLen,
283                     dlyAmt = instr.i[24] / 255,
284                     dly = instr.i[25] * rowLen;
285 
286                 // Calculate start sample number for this row in the pattern
287                 rowStartSample = (p * patternLen + row) * rowLen;
288 
289                 // Generate notes for this pattern row
290                 for (col = 0; col < 4; ++col) {
291                     n = cp ? instr.c[cp - 1].n[row + col * patternLen] : 0;
292                     if (n) {
293                         if (!noteCache[n]) {
294                             noteCache[n] = createNote(instr, n);
295                         }
296 
297                         // Copy note from the note cache
298                         var noteBuf = noteCache[n];
299                         for (j = 0, i = rowStartSample * 2; j < noteBuf.length; j++, i += 2) {
300                           chnBuf[i] += noteBuf[j];
301                         }
302                     }
303                 }
304 
305                 // Perform effects for this pattern row
306                 for (j = 0; j < rowLen; j++) {
307                     // Dry mono-sample
308                     k = (rowStartSample + j) * 2;
309                     rsample = chnBuf[k];
310 
311                     // We only do effects if we have some sound input
312                     if (rsample || filterActive) {
313                         // State variable filter
314                         f = fxFreq;
315                         if (fxLFO) {
316                             f *= oscLFO(lfoFreq * k) * lfoAmt + 0.5;
317                         }
318                         f = 1.5 * Math.sin(f);
319                         low += f * band;
320                         high = q * (rsample - band) - low;
321                         band += f * high;
322                         rsample = fxFilter == 3 ? band : fxFilter == 1 ? high : low;
323 
324                         // Distortion
325                         if (dist) {
326                             rsample *= dist;
327                             rsample = rsample < 1 ? rsample > -1 ? osc_sin(rsample*.25) : -1 : 1;
328                             rsample /= dist;
329                         }
330 
331                         // Drive
332                         rsample *= drive;
333 
334                         // Is the filter active (i.e. still audiable)?
335                         filterActive = rsample * rsample > 1e-5;
336 
337                         // Panning
338                         t = Math.sin(panFreq * k) * panAmt + 0.5;
339                         lsample = rsample * (1 - t);
340                         rsample *= t;
341                     } else {
342                         lsample = 0;
343                     }
344 
345                     // Delay is always done, since it does not need sound input
346                     if (k >= dly) {
347                         // Left channel = left + right[-p] * t
348                         lsample += chnBuf[k-dly+1] * dlyAmt;
349 
350                         // Right channel = right + left[-p] * t
351                         rsample += chnBuf[k-dly] * dlyAmt;
352                     }
353 
354                     // Store in stereo channel buffer (needed for the delay effect)
355                     chnBuf[k] = lsample | 0;
356                     chnBuf[k+1] = rsample | 0;
357 
358                     // ...and add to stereo mix buffer
359                     mMixBuf[k] += lsample | 0;
360                     mMixBuf[k+1] += rsample | 0;
361                 }
362             }
363         }
364 
365         // Next iteration. Return progress (1.0 == done!).
366         mCurrentCol++;
367         return mCurrentCol / 8;
368     };
369 
370     // Create a WAVE formatted Uint8Array from the generated audio data
371     this.createWave = function() {
372         // Create WAVE header
373         var l1 = mNumWords * 2 - 8;
374         var l2 = l1 - 36;
375         var headerLen = 44;
376         var wave = new Uint8Array(headerLen + mNumWords * 2);
377         wave.set(
378             [82,73,70,70,
379              l1 & 255,(l1 >> 8) & 255,(l1 >> 16) & 255,(l1 >> 24) & 255,
380              87,65,86,69,102,109,116,32,16,0,0,0,1,0,2,0,
381              68,172,0,0,16,177,2,0,4,0,16,0,100,97,116,97,
382              l2 & 255,(l2 >> 8) & 255,(l2 >> 16) & 255,(l2 >> 24) & 255]
383         );
384 
385         // Append actual wave data
386         for (var i = 0, idx = headerLen; i < mNumWords; ++i) {
387             // Note: We clamp here
388             var y = mMixBuf[i];
389             y = y < -32767 ? -32767 : (y > 32767 ? 32767 : y);
390             wave[idx++] = y & 255;
391             wave[idx++] = (y >> 8) & 255;
392         }
393 
394         // Return the WAVE formatted typed array
395         return wave;
396     };
397 
398     // Get n samples of wave data at time t [s]. Wave data in range [-2,2].
399     this.getData = function(t, n) {
400         var i = 2 * Math.floor(t * 44100);
401         var d = new Array(n);
402         for (var j = 0; j < 2*n; j += 1) {
403             var k = i + j;
404             d[j] = t > 0 && k < mMixBuf.length ? mMixBuf[k] / 32768 : 0;
405         }
406         return d;
407     };
