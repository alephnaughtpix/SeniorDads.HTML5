460         function truColourToCanvasColour(truColourVal) {
461             //Value is in format: RRRRrGGGGGgBBBBb
462             //& %1111100000000000 >> 8 = %RRRRR000 -> 0-0xff - 8
463             //& %0000011111100000 >> 3 = %GGGGGG00 -> 0-0xff - 3
464             //& %0000000000011111 << 3 = %BBBBB000 -> 0-0xff - 8
465             var r = ((truColourVal & 63488) >> 8);
466             var g = ((truColourVal & 2016) >> 3);
467             var b = ((truColourVal & 31) << 3);
468             return [r, g, b, 0xff];
469         }

510     function truColourToWebColour(truColourVal) {
511         //Value is in format: RRRRrGGGGGgBBBBb
512         //& %1111100000000000 >> 12 = %RRRR -> 0-0xf
513         //& %0000011111100000 >> 7  = %GGGG -> 0-0xf
514         //& %0000000000011111 >> 1  = %BBBB -> 0-0xf
515         var r = ((truColourVal & 63488) >> 12).toString(16);
516         var g = ((truColourVal & 2016) >> 7).toString(16);
517         var b = ((truColourVal & 31) >> 1).toString(16);
518         return '#' + r + r + g + g + b + b;
519     }
