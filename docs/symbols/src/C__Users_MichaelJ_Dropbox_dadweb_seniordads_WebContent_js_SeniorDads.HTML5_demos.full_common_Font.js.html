 12 SeniorDads.CreateNameSpace("SeniorDads.Font.Tos");
 13 /**
 14  * <h1>SeniorDads.Font.Tos</h1>
 15  * <p><strong>{@class}</strong></p>
 16  * <p><em>var myFont = new SeniorDads.Font.Tos(url,graphicsMode);</em></p>
 17  * <p>This allows for printing text in an authentic Atari TOS font, and in "lo-res" and "med-res" modes.</p>
 18  * @class Atari TOS font drawing utility
 19  * @param {string} url URL of the font file.
 20  * @param {number} [graphicsMode=0] Text mode. 0 = lo-res (40 x 24), 1 = med-res (80 x 24).
 21  * @property {number} width Text width of screen.
 22  * @property {number} height Text height of screen.
 23  * @property {boolean} loaded Whether the resource has been loaded.
 24  * @property {number} resource_size Size of resource
 25  * @property {number} resource_loaded Resource to be loaded
 26  */
 27 SeniorDads.Font.Tos = function (url, graphicsMode) {
 28     if (graphicsMode === undefined) graphicsMode = 0;				// Default value if not set.
 29     var fontData, fontBuffer, width, height, x, y, loader;
 30     var foreground = [0, 0, 0, 0xff];								// Standard foreground and background colours.
 31     var background = [0xff, 0xff, 0xff, 0xff];
 32     var loaded = false;												// Start off assuming we haven't loaded the font file.
 33     // Set up canvas for single font character
 34     var characterCanvas = document.createElement('canvas');
 35     characterCanvas.id = "SeniorDads.Font.Tos.Store";
 36     characterCanvas.width = 8;
 37     characterCanvas.height = 8;
 38     characterCanvas.style.visibility = 'hidden';
 39     var charCanvasContext = characterCanvas.getContext('2d');
 40     // Set text mode
 41     setTextMode(graphicsMode);
 42     // Start loading the font
 43     loader = new SeniorDads.Loader.Binary(url, fontData, processFont);
 44 
 45     Object.defineProperties(this, {
 46     	loaded:          { get: function() { return loaded; } },			// Whether the font file has loaded.
 47     	resource_size:   { get: function() { return 1; } },					// Size of the resource.
 48     	resource_loaded: { get: function() { return (loaded) ? 1 : 0; } },	// The amount already loaded.
 49     	width:           { get: function() { return width; } },				// Text width of screen.
 50     	height:          { get: function() { return height; } } 			// Text height of screen.
 51     } );
 52 
 53     /**
 54      * <p>Set graphics/text mode.</p>
 55      * <p>The Atari ST has the following graphics modes:
 56      * <ul>
 57      * <li>Mode 0: 16 colours, 320 x 200 pixels, 40 x 24 text</li>
 58      * <li>Mode 1: 4 colours, 640 x 200 pixels, 80 x 24 text</li>
 59      * <li>Mode 2: Monochrome, 640 x 400 pixels, 80 x 24 text</li>
 60      * </ul>
 61      * <p>So far, the text display for modes 0-1 have been emulated here. Mode 2, which 
 62      * requires a different font, will come in future.</p>
 63      * @function
 64      * @param {number} [setGraphicsMode=0] Text mode. 0 = lo-res (40 x 24), 1 = med-res (80 x 24).
 65      */
 66     this.SetTextMode = setTextMode;
 67     /**
 68      * Set text colours to [r,g,b,a] colours.
 69      * @function
 70      * @param {UInt8Array[4]} myForeground Foreground colour
 71      * @param {UInt8Array[4]} myBackground Background colour
 72      * @example
 73      *  var myFont = new SeniorDads.Font.Tos("resources/tosfont.bin");
 74      *  myFont.setColours(
 75      *  	[0xff, 0xff, 0, 0xff], 	// yellow foreground, opaque
 76      *  	[0, 0, 0, 0xcc]			// blue background, slightly transparent
 77      *  )
 78      *  myFont.printText(myScreen, 0, 0, "Yellow text on translucent blue!");
 79      */
 80     this.SetColours = setColours;
 81     /**
 82      * <p>Print text on {@link SeniorDads.ScreenHandler.Screen} object.</p>
 83      * <p>The following VT52 codes can be used within the text. (Case sensitive.)</p>
 84      * <ul>
 85      * <li><strong>chr(27)+'p'</strong> - invert text colours.
 86      * <li><strong>chr(27)+'q'</strong> - revert text colours.
 87      * <li><strong>chr(27)+'Y'+chr(x+32)+chr(y+32)</strong> - position following characters at text coordinates (x,y) .
 88      * @function
 89      * @param {SeniorDads.ScreenHandler.Screen} screen Screen to write text to.
 90      * @param {number} px *Text* x-coordinate to print at.
 91      * @param {number} py *Text* y-coordinate to print at.
 92      * @param {string} text Text to print.
 93      */
 94     this.PrintText = printText;
 95 
 96     /*
 97      * When the font has been loaded, process into a font buffer.
 98      * 
 99      * The font buffer is composed of a hash which allows easy access by ASCII character.
100      * Each element of the hash contains an array of boolean values representing each pixel 
101      * of the font for a particular character.
102      */
103     function processFont(fontData) {
104         fontBuffer = new Object;						// Initalise the font buffer
105         var ascii_start = " ".charCodeAt(0);			// The font data starts at " "
106         var ascii_end = "~".charCodeAt(0) - 1;			// ... and ends at "~".
107         var fontDataView = new DataView(fontData);		// Set up view of binary font data
108         var fontDataPointer = 0;
109         var currentChar;
110         for (var i = ascii_start; i <= ascii_end; i++) { 	// Start iterating through the characters of the font
111             currentChar = String.fromCharCode(i);
112             fontBuffer[currentChar] = new Array(8 * 8);		// Set up array in hash for the current character
113             for (var j = 0; j < 8; j++) {					// Font character is 8 bytes long, and eachbyte represents a pixel row
114                 var currentCharLine = fontDataView.getUint8(fontDataPointer++);	// Get a byte 
115                 for (var k = (8 - 1); k > -1; --k)			// For each of the 8 pixels, set true or false in the font buffer
116                     fontBuffer[currentChar][(j * 8) + (8 - k)] = ((currentCharLine & (1 << k)) > 0);
117             }
118         }
119         // Now the font is processed, we can flag the resource as loaded.
120         loaded = true;
121     }
122 
123     /*
124      * Set Text mode.
125      * 
126      * 0 = lo-res (40 x 24), 1 = med-res (80 x 24).
127      * 
128      * There's not really too much to this. However, there may be more when Hi-res (Mode 2) is implmented.
129      */
130     function setTextMode(setGraphicsMode) {
131         setGraphicsMode = (typeof setGraphicsMode === "undefined") ? 0 : setGraphicsMode; 			// if not set, assume mode 0
132         setGraphicsMode = ((0 > setGraphicsMode) || (setGraphicsMode > 1)) ? 0 : setGraphicsMode; 	// If outside 0-1 range, assume mode 0
133         // Set text width and height according to mode.
134         switch (setGraphicsMode) {
135             case 0:
136                 width = 40;
137                 height = 24;
138                 break;
139             case 1:
140                 width = 80;
141                 height = 24;
142                 break;
143         }
144     }
145 
146     /*
147      * Set text colours
148      */
149     function setColours(myForeground, myBackground) {
150         foreground = myForeground;
151         background = myBackground;
152     }
153 
154     /*
155      * Print text onto screen.
156      */
157     function printText(screen, px, py, text) {
158         var context = screen.context;
159         var ctrl = String.fromCharCode(27);		// Control code start for VT52 
160         var lines = text.split('\n');			// Split into lines by carriage return.
161         var linecount = lines.length;
162         var gfxX = px * 8;						// Convert text coordinates into canvas coordinates.
163         var gfxY = py * 8;
164         var h, i, j, length, charBuffer, canvasDataPosition, charBufferPointer, colour, currentLine;
165         for (h = 0; h < linecount; h++) {		// Go through each line
166             currentLine = lines[h];				// For each line...
167             length = currentLine.length;
168             for (i = 0; i < length; i++) {		// ... go through each character
169                 if (currentLine.charAt(i) == ctrl) {	// If VT52
170                     i++;								// .. next is control code
171                     switch (currentLine.charAt(i++)) {
172                     case "p":							// ESC p - invert text colours
173                         invertTextColours();
174                         break;
175                     case "q":							// ESC q - Un-invert text colours
176                         invertTextColours();
177                         break;
178                     case "Y":							// ESC Y - Position text
179                         py = currentLine.charCodeAt(i++) - 32;	// Get text coords from next two characters
180                         px = currentLine.charCodeAt(i++) - 32;  // Coords are expressed as chr(x+32)+chr(y+32)
181                         gfxX = px * 8;							// Convert text coordinates into canvas coordinates.
182                         gfxY = py * 8;
183                         break;
184                     }
185                     i--;
186                 } else {
187                     charBuffer = fontBuffer[currentLine.charAt(i)];		// Get the font buffer for the character to print.
188                     var canvasData = context.createImageData(8, 8);		// Set up image data for character.
189                     canvasDataPosition = 0;
190                     charBufferPointer = 0;
191                     for (j = 0; j < (8 * 8); j++) {						// For each of the pixels
192                     	// Set the colour according whether the pixel is on.
193                         colour = (charBuffer[charBufferPointer++]) ? foreground : background;
194                         canvasData.data[canvasDataPosition++] = colour[0];		// Now plot it.
195                         canvasData.data[canvasDataPosition++] = colour[1];
196                         canvasData.data[canvasDataPosition++] = colour[2];
197                         canvasData.data[canvasDataPosition++] = colour[3];
198                     }
199                     charCanvasContext.putImageData(canvasData, 0, 0);			// Draw the character on screen.
200                     context.drawImage(characterCanvas, gfxX, gfxY);
201                     gfxX += 8;		// Go onto next character
202                     px++;
203                 }
204             }
205             gfxY += 8;		// Now we're at the end of line, go onto the next line.
206             py++;
207             gfxX = 0;
208             px = 0;
209         }
210         x = px;
211         y = py;
212     }
213 
214     /*
215      * Invert text colours
216      */
217     function invertTextColours() {
218         var temp = foreground;
219         foreground = background;
220         background = temp;
221     }
222 };
