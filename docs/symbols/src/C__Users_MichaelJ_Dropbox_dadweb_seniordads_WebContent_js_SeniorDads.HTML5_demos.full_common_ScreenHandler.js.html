 12 /**
 13  * <h1>SeniorDads.ScreenHandler</h1>
 14  * <p><strong>{@class}</strong></p>
 15  * <p><em>var myScreenHandler = new SeniorDads.ScreenHandler(element, name, width, height);</em></p>
 16  * <p>This is a screen handler which handles the creation and usage of canvas screens.</p>
 17  * <p>From here, individual screens can be added via {@link SeniorDads.ScreenHandler.Screen}.</p>
 18  * @class Senior Dads screen handler
 19  * @param {DOM Object} element HTML DOM element to place the screens in.
 20  * @param {string} name Unique name ID for the screen handler.
 21  * @param {number} width Width of the screen.
 22  * @param {number} height Height of the screen.
 23  * @param {string} [bgColour] Background colour of screen handler.
 24  */
 25 SeniorDads.ScreenHandler = function (element, name, width, height, bgColour) {
 26 	if (bgColour===undefined) bgColour = '#000000';					// Default background colour is black
 27     var screenWrapper = document.getElementById(name);
 28     if (screenWrapper!=null) 
 29     	screenWrapper.parentNode.removeChild(screenWrapper);
 30     screenWrapper = element.appendChild(document.createElement('div'));
 31     var orderPointer = 1;
 32     var screenList = new Object;
 33     var imageData;
 34     var waitVblAnim;
 35     screenWrapper.id = name;
 36     screenWrapper.style.width = width + "px";
 37     screenWrapper.style.height = height + "px";
 38     screenWrapper.style.backgroundColor = bgColour;
 39     screenWrapper.style.position = "relative";
 40 
 41     /**
 42      * Add a {@link SeniorDads.ScreenHandler.Screen} to screen handler
 43      * @function
 44      * @param {string} name Unique name ID for the screen.
 45      * @param {number} scale Scale of screen.
 46      * @param {SeniorDads.Image} image {@link SeniorDads.Image}
 47      */
 48     this.add = add;
 49 
 50     /**
 51      * Wait for a number of frames (This will need updating, as it's not very accurate!)
 52      * @function
 53      * @param {number} noOfVBLs Number of frames.
 54      * @param {function} callback Function call at end of wait
 55      */
 56     this.WaitVbl = waitVbl;
 57     /**
 58      * Stop frame wait
 59      * @function
 60      */
 61     this.FlushWaitVbl = flushWaitVbl;
 62     /**
 63      * Set screen handler to go fullscreen.
 64      * @function
 65      */
 66     this.fullscreen = goFullScreen;
 67 
 68     function add(myName, scale, image) {
 69         screenList[name] = new SeniorDads.ScreenHandler.Screen(screenWrapper, myName, width, height, scale, orderPointer++, image);
 70         return screenList[name];
 71     }
 72 
 73     function waitVbl(noOfVBLs, callback) {
 74         waitVblAnim = setTimeout(function () {
 75         	window.requestAnimFrame(callback);
 76         }, noOfVBLs * 20);
 77     }
 78 
 79     function flushWaitVbl() {
 80         clearTimeout(waitVblAnim);
 81     }
 82         
 83     function goFullScreen() {
 84 		if (element.requestFullscreen) {
 85 			element.requestFullscreen();
 86 		} else if (element.mozRequestFullScreen) {
 87 			element.mozRequestFullScreen();
 88 		} else if (element.webkitRequestFullScreen) {
 89 			element.webkitRequestFullScreen(Element.ALLOW_KEYBOARD_INPUT);
 90 		}
 91 	}

 94     SeniorDads.ScreenHandler.Screen = function (element, name, width, height, scale, zIndex, image) {
 95         if (image === undefined) image = null;
 96         var loaded = false;
 97         var backgroundColour;
 98         var canvas = element.appendChild(document.createElement('canvas'));
 99         canvas.id = name;
100         var scaledWidth = width * scale;
101         var scaledHeight = height * scale;
102         canvas.width = scaledWidth;
103         canvas.height = scaledHeight;
104         canvas.style.position = "absolute";
105         canvas.style.top = "0px";
106         canvas.style.left = "0px";
107         canvas.style.zIndex = zIndex;
108         var context = canvas.getContext('2d');
109         context.setTransform(1, 0, 0, 1, 0, 0); // Reset scaling
110         context.scale(scale, scale);            // Set scaling 
111         context.imageSmoothingEnabled = false;  // Switch off interpolation
112         hide();
113 
114         if (image != null)
115             if (image.loaded) {
116                 drawImage();
117             } else {
118                 image.loadedCallback = drawImage;
119             }
120         else
121             loaded = true;
122 
123 //        this.loaded = function () { return loaded; };
124 //        this.resource_size = function () { return 1; };					// Size of the resource
125 //        this.resource_loaded =  function () { return 0; }; 				// The amount already loaded (we get this from the instance of the class we're wrapping.)
126 //        this.canvas = function () { return canvas; };
127         //this.context = function () { return context; };
128 //        this.name = function () { return name; };
129 //        this.zIndex = function () { return zIndex; };
130 //        this.width = function () { return width; };
131         //this.height = function () { return height; };
132 //        this.scale = function () { return scale; };
133         Object.defineProperties(this, {
134            	loaded:           { get: function() { return loaded; } },							 // Whether the screen has loaded.
135            	resource_size:    { get: function() { return 1; } },								 // Size of the resource.
136            	resource_loaded:  { get: function() { return (loaded) ? 1 : 0; } },					 // The amount already loaded.
137         	canvas: {
138             	get: function() { return canvas; }
139             },
140         	name: {
141             	get: function() { return name; }
142             },
143         	zIndex: {
144             	get: function() { return zIndex; }
145             },
146         	height: {
147             	get: function() { return height; }
148             },
149         	width: {
150             	get: function() { return width; }
151             },
152         	scale: {
153             	get: function() { return height; }
154             },
155         	context: {
156             	get: function() { return context; }
157             }
158         } );
159 
160         this.Show = show;
161         this.Hide = hide;
162         this.Clear = clear;
163 
164         function drawImage() {
165             if (!(image == null)) {
166                 imageData = image.imageData;
167                 context.drawImage(imageData, 0, 0);
168                 loaded = true;
169             }
170         }
171 
172         function clear(myColour) {
173             if (myColour === undefined)
174                 myColour = backgroundColour;
175             else
176                 backgroundColour = myColour;
177             if (myColour != null) {
178                 context.fillStyle = myColour;
179                 context.fillRect(0, 0, width, height);
180             }
181         }
182 
183         function show() {
184             canvas.style.visibility = 'visible';
185         }
186 
187         function hide() {
188             canvas.style.visibility = 'hidden';
189         }
190     };

204         if (name === undefined)

206         else {
207         	codefCanvas = new canvas(width,height,name);
208 	        codefCanvas.canvas.style.position = "absolute";

221             },
