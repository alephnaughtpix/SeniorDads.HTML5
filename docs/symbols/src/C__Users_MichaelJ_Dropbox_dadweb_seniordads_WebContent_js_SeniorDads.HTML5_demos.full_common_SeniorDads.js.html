 38 // create a general purpose namespace method
 39 // this will allow us to create namespace a bit easier
 40 SeniorDads.CreateNameSpace = function (namespace) {
 41     var nsparts = namespace.split(".");
 42     var parent = SeniorDads;
 43  
 44     // we want to be able to include or exclude the root namespace 
 45     // So we strip it if it's in the namespace
 46     if (nsparts[0] === "SeniorDads") {
 47         nsparts = nsparts.slice(1);
 48     }
 49  
 50     // loop through the parts and create 
 51     // a nested namespace if necessary
 52     for (var i = 0; i < nsparts.length; i++) {
 53         var partname = nsparts[i];
 54         // check if the current parent already has 
 55         // the namespace declared, if not create it
 56         if (typeof parent[partname] === "undefined") {
 57             parent[partname] = {};
 58         }
 59         // get a reference to the deepest element 
 60         // in the hierarchy so far
 61         parent = parent[partname];
 62     }
 63     // the parent is now completely constructed 
 64     // with empty namespaces and can be used.
 65     return parent;
 66 };

 74 /**
 75  * <h1>Wobbler</h1>
 76  * <p><strong>Create a "wobble" effect.</strong></p>
 77  * <p><em>var myWobbler = new SeniorDads.Wobbler(hWobbleParameters, vWobbleParameters, zWobbleParameters);</em></p>
 78  * @constructor
 79  * @class Create a "wobble" effect.
 80  * @param {Object[]} hWobbleParameters Horizontal wobble parameters.
 81  * @param {Object[]} [vWobbleParameters] vertical wobble parameters.
 82  * @param {Object[]} [zWobbleParameters] Z-index wobble parameters.
 83  * @property {number} h Get next horizontal wobble differential.
 84  * @property {number} v Get next vertical wobble differential.
 85  * @property {number} z Get next Z-index wobble differential.
 86  * @example
 87  *  //Place an image in the center of a 640x400 area and wobble it.
 88  *  
 89  *  // First create the area and load the image
 90  *  var myPlaceholder = document.getElementById('placeholder'); // <- Assuming we have a div with ID of 'placeholder' on the web page.
 91  *  myPlaceholder.style.width = "640px";
 92  *  myPlaceholder.style.height = "400px";
 93  *  myPlaceholder.style.position = "relative";
 94  *  var myImage = new Image();
 95  * 	myImage = "myImage_url.png";
 96  * 	myImage.style.position = "absolute";
 97  *  myPlaceholder.appendChild(myImage);
 98  *  
 99  *  var myWobbler;
100  *  
101  *  // When the image is loaded, create the wobbler
102  *  myImage.onload = function() {
103  *  
104  *  	myWobbler = new SeniorDads.Wobbler(
105  *  		[ 
106  *  			{value: 0, amp: 3, inc: 0.30},
107  *  			{value: 0.5, amp: 3, inc: 0.40}
108  *  		],
109  *  		[
110  *  			{value: 0.5, amp: 3, inc: 0.20},
111  *  			{value: 0, amp: 2, inc: 0.10}
112  *  		]);
113  *  	// Start the animation loop.
114  *  	animationLoop();
115  *  }
116  *  
117  *  // Animation loop- wobbles the image relative to the centre of the area.
118  *  function animationLoop() {
119  *  	var x = 320 + myWobbler.h();
120  *  	var y = 200 + myWobbler.v();
121  *  	myImage.style.left = x + "px";
122  *  	myImage.style.top = y + "px";
123  *  	requestAnimFrame( animationLoop );
124  *  }
125  */
126 SeniorDads.Wobbler = function (hWobbleParameters, vWobbleParameters, zWobbleParameters) {
127 	if (hWobbleParameters === undefined) hWobbleParameters = null;
128 	if (vWobbleParameters === undefined) vWobbleParameters = null;	// Default for optional parameter
129 	if (zWobbleParameters === undefined) zWobbleParameters = null;	// Default for optional parameter
130 	
131 	// Set properties
132 	this.h = function() { return wobble(hWobbleParameters); };		// get next horizontal wobble differential
133 	// Get next vertical wobble differential
134 	if ( ( hWobbleParameters != null) && (vWobbleParameters == null) )
135 		this.v = this.h;											// If only "horizontal" wobble supplied, vertical differential = horizontal
136 	else
137 	// Get next z-index wobble differential
138 	this.v = function() { return wobble(vWobbleParameters); };
139 	if ( ( hWobbleParameters != null) && (zWobbleParameters == null) )
140 		this.z = this.h;											// If only "horizontal" wobble supplied, z-index differential = horizontal
141 	else
142 		this.z = function() { return wobble(zWobbleParameters); };
143 		
144 	/* Main wobbling calculator.
145 	 * 
146 	 * This combines the sines waves and their positions as defined in the 
147 	 * wobble parameters to arrive at a combined wobble value.
148 	 * 
149 	 */
150 	function wobble(wobbleParameters) {
151 		var wobbleVal = null;			// Default value for invalid wobble.
152 		if (wobbleParameters != null)	// only calculate if we have wobble parameters.
153 			wobbleVal = 0;				// Reset wobble value.
154 			for( var i = 0; i < wobbleParameters.length; i++) { 							// For each wobble param...
155 				wobbleVal += Math.sin(wobbleParameters[i].value) * wobbleParameters[i].amp; // Add current sine wave position of wobble element
156 				wobbleParameters[i].value += wobbleParameters[i].inc;						// Move onto next sine wave position.
157 			}
158 		// Combined wobble value has now been calculated.
159 		return wobbleVal; 
160 	}
161 	
162 };

164 /**
165  * <h1>Bomb</h1>
166  * <p><strong>Display the Atari TOS "bomb" pattern for exceptions in Atari 16/24bit machines.</strong></p>
167  * <p><em>var myBomb = new SeniorDads.Bomb();</em></p>
168  * @class Display the Atari TOS "bomb" pattern.
169  * @constructor
170  * @example
171  *  // First create the canvas to bomb.
172  *  var myPlaceholder = document.getElementById('placeholder'); // <- Assuming we have a div with ID of 'placeholder' on the web page.
173  *  var myCanvas = myPlaceholder.appendChild(document.createElement('canvas'));
174  *  
175  *  // Now bomb it!
176  *  var myBomb = new SeniorDads.Bomb();
177  *  myBomb.bombCanvas(myCanvas, 4);
178  */
179 SeniorDads.Bomb = function () {
180     // Atari TOS bomb bitmap pattern- copied from old demo source code, itself copied from ROM listing in ST Internals!
181     var bombBits = new Uint16Array([
182         1536,  // %0000011000000000
183         10496, // %0010100100000000
184         128,   // %0000000010000000
185         18496, // %0100100001000000
186         4592,  // %0001000111110000
187         496,   // %0000000111110000
188         2044,  // %0000011111111100
189         4094,  // %0000111111111110
190         4094,  // %0000111111111110
191         8191,  // %0001111111111111
192         8175,  // %0001111111101111
193         4078,  // %0000111111101110
194         4062,  // %0000111111011110
195         2044,  // %0000011111111100
196         1016,  // %0000001111111000
197         224    // %0000000011100000
198     ]);
199     // Set up a canvas for the canvas friendly version of the bomb image.
200     var truColBombCanvas = document.createElement('canvas');
201     truColBombCanvas.id = "SeniorDads.Bomb.Store";
202     truColBombCanvas.width = 16;					// Bomb is 16x16 pixels.
203     truColBombCanvas.height = 16;
204     truColBombCanvas.style.visibility = 'hidden';
205 
206     convertBombToTruCol();			// Convert the bitmap to the canvas friendly version of the bomb image.
207 
208     /**
209      * <h3>bombCanvas</h3>
210      * 
211      * <p><strong>Display bombs.</strong></p>
212      * <p>The bombs are drawn in black on white in the middle left of the screen.</p>
213      * @function
214      * @param {SeniorDads.ScreenHandler.Screen} screen Screen to draw to.
215      * @param {number} bombCount Number of bombs to draw
216      */
217     this.bombCanvas = bombCanvas;	// Setup method for creating bombs
218 
219     /*
220      * Convert bomb bitmap to canvas colours.
221      * 
222      * For each pixel we set the foreground/background colour accordingly. 
223      * ie black on white, as used in the Atari Falcon in trucolor mode.
224      *  
225      */
226     function convertBombToTruCol() {
227         var foreground = [0, 0, 0, 0xff];					// Foreground is black.
228         var background = [0xff, 0xff, 0xff, 0xff];			// Background is white.
229         var context = truColBombCanvas.getContext('2d');
230         var canvasData = context.createImageData(16, 16);	// Create the 16x16 image data.
231         var colour;
232         var canvasDataPosition = 0;							// Start at the top left of the image data.
233         for (var i = 0; i < 16; i++) {						// There's 16 words in the bomb,
234             var currentBombLine = bombBits[i];				// ... each of which equals a single row of the bomb pixels,
235             for (var j = (16 - 1); j > -1; --j) {			// ... which is 16 pixels in length
236             	// Bit-test current pixel in word to determine if there's a pixel present.
237             	// If there is use the foreground colour, otherwise the background colour.
238                 colour = ((currentBombLine & (1 << j)) > 0) ? foreground : background;
239             	// Now plot the pixel in the relevant colour.
240                 canvasData.data[canvasDataPosition++] = colour[0];
241                 canvasData.data[canvasDataPosition++] = colour[1];
242                 canvasData.data[canvasDataPosition++] = colour[2];
243                 canvasData.data[canvasDataPosition++] = colour[3];
244             }
245         }
246         // Now the bomb is drawn, copy it to the bomb canvas.
247         context.putImageData(canvasData, 0, 0);
248     }
249     
250     /*
251      * bombCanvas
252      * # screen - SeniorDads.ScreenHandler.Screen to draw to.
253      * # bombCount - number of bombs to draw
254      */
255     function bombCanvas( screen, bombCount ) {
256     	var y, context;
257     	var x = 0;
258     	y = (screen.height - 16) / 2;			// Go to middle left of screen
259     	context = screen.context;				// Get canvas context of screen
260     	for (var i=0; i < bombCount; i++) {				// For the defined number of bombs...
261     		context.drawImage(truColBombCanvas, x, y);	// Copy the bomb canvas to screen.
262     		x += 16;									// Move to position of next bomb.
263     	} 
264     };
265 };

274 /**
275  * <h1>SeniorDads.Demos.Demo</h1>
276  * <p><strong>Senior Dads demo framework.</strong></p>
277  * <p><em>var myDemo = new SeniorDads.Demos.Demo(preLoader, loader, init, abort);</em></p>
278  * <p>This class allows you to add individual screens (See {@link SeniorDads.Demos.Demo.DemoPart}), and to handle
279  * general loading of resources, and initialisation before starting the demo. There is also a <i>preloader</i>, which can be
280  * used if you need to set something up before loading resources. (eg a "Now loading demo" type screen.)</p>
281  * <p>Once the demo is started, the class can handle moving through the list of screens, or stopping the demo altogether.</p>
282  * <p>Both the preloader and loaders can be supplied with callback functions to be executed when they're done.<p>
283  * @constructor 
284  * @class Senior Dads demo framework
285  * @param {function} preLoader Function to call <i>before</i> loading resources
286  * @param {function} loader Function to call to <i>start</i> loading resources
287  * @param {function} init Function to call after loading resources but <i>before</i> running the demo.
288  * @param {function} abort Function to call if the demo needs to make a speedy exit.
289  * @param {function} [onEnd] Function to call when the demo has fully ended. (ie after abort)
290  * @property {bool} started Has the demo started?
291  * @property {bool} ended Has the demo ended?
292  */
293 SeniorDads.Demos.Demo = function(preLoader, loader, init, abort, onEnd) {
294 	if (onEnd === undefined) onEnd = null;
295 	var currentDemoPart;			// Currently running SeniorDads.Demos.Demo.DemoPart 
296 	var demoPartList = [];			// Array of SeniorDads.Demos.Demo.DemoPart containing the demo screens.
297 	var currentDemoPartCounter = 0; // Current position of demo
298 	var started = false;			// Whether the demo has started.
299 	var ended = false;				// Whether the demo has ended.
300     
301 	/**
302 	 * Preload a demo, without initalising or starting.
303 	 * @function
304 	 * @param {function} [preLoaderDone] Function to call when preloader is finished.
305 	 */
306     this.preload = preLoad;
307     
308 	/**
309 	 * Load a demo, without initalising or starting.
310 	 * @function
311 	 * @param {function} [loaderDone] Function to call when loader is finished.
312 	 */
313     this.load = load;
314 	/**
315 	 * Initialise a demo, without starting.
316 	 * @function
317 	 */
318     this.init = initialise; 
319 	/**
320 	 * Abort a demo.
321 	 * @function
322 	 */
323     this.abort = abortDemo;
324 	/**
325 	 * Add a screen to a demo. This requires creation of a {@link SeniorDads.Demos.Demo.DemoPart}.
326 	 * @example
327 	 *  // Example stub for a simple one screen demo.
328 	 *  
329 	 *  // Create the demo.
330 	 *  var myDemo = new SeniorDads.Demos.Demo(preLoader, loader, init, abort);
331 	 *  // Add a screen to it.
332 	 *  myDemo.add( new SeniorDads.Demos.Demo.DemoPart( part1init, part1Start, part1abort, demoEnded );
333 	 *  // Load and start the demo.
334 	 *  myDemo.loadAndStart();
335 	 *  
336 	 *  // [...]
337 	 *  
338 	 *  // PRELOADER:
339 	 *  // This is run first by the demo object, with a callback to the 
340 	 *  // loader function in the parameter "preLoaderDone".
341 	 *  function preLoader( preLoaderDone ) {
342 	 *  	// [... Pre loading code ...]
343 	 *  	preLoaderDone();	// <- This will call the function "loader" below.
344 	 *  }
345 	 *  
346 	 *  // LOADER:
347 	 *  // This is run second as a result of the preloader callback. It contains
348 	 *  // a callback to the initalise and start function in the parameter "loaderDone".
349 	 *  function loader( loaderDone ) {
350 	 *  	// [... Loading code ...]
351 	 *  	loaderDone() 	// <- This will call the initialise and start functions in the object.
352 	 *  }
353 	 *  
354 	 *  // INITIALISE:
355 	 *  // This is run third by the demo object's initialise and start functions.
356 	 *  //
357 	 *  // After this is run, the first demo screen is run (ie function 'part1Start'.)
358 	 *  function init() {
359 	 *  	// [... Initalisation code ...]
360 	 *  }
361 	 *  
362 	 *  // ABORT:
363 	 *  // This is called if "myDemo.abort();" is used.
364 	 *  function abort() {
365 	 *  	// [... Demo abort code ...]
366 	 *  }
367 	 *  
368 	 *  // DEMO SCREEN INITIALISE:
369 	 *  // This is run when the demo screen is first added to the demo. 
370 	 *  // (ie before the start of the demo, and as a result of the command 
371 	 *  // "myDemo.add( ... )".)
372 	 *  function part1init() {
373 	 *  	// [... Screen initialisation code ...]
374 	 *  }
375 	 *  
376 	 *  // DEMO SCREEN RUN:
377 	 *  // This is run when the demo object runs the first screen.
378 	 *  function part1Start() {
379 	 *  	// [... Screen main code ...]
380 	 *  }
381 	 *  
382 	 *  // DEMO SCREEN ABORT:
383 	 *  // This is called if "myDemo.abort();" is used.
384 	 *  function part1abort() {
385 	 *  	// [... Screen abort code ...]
386 	 *  }
387 	 *  
388 	 *  // DEMO ENDED:
389 	 *  // Called when the demo has fully finished.
390 	 *  function demoEnded() {
391 	 *  	// [... This demo is ended! ...]
392 	 *  }
393 	 * @function
394 	 * @param {SeniorDads.Demos.Demo.DemoPart} currentDemo Demo screen.
395 	 */
396     this.add = addDemoPart;
397 	/**
398 	 * Start demo.
399 	 * @function
400 	 */
401     this.start = start;
402 	/**
403 	 * Go to next demo screen.
404 	 * @function
405 	 */
406     this.next = nextDemoPart;
407 	/**
408 	 * Load, initialise and start demo. This will call, in the following order:
409 	 * <ul>
410 	 * <li>the preloader,</li>
411 	 * <li>the loader,</li>
412 	 * <li>the initialisation,</li>
413 	 * <li>and finally, the demo, from the first screen onwards.</li>
414 	 * @function
415 	 */
416     this.loadAndStart = loadAndStart;
417 
418     Object.defineProperties(this, {
419        	started: { get: function() { return started; } },	// Whether the demo has started
420        	ended:   { get: function() { return ended; } }		// Whether the demo has ended
421     } );
422 
423     
424     this.currentPart = function() { 
425     	return currentDemoPart; 
426     	};

429     /*
430      * Add demo part.
431      * 
432      * This pushes the demo screen in "currentDemo" onto the array of SeniorDads.Demos.Demo.DemoPart
433      */
434     function addDemoPart(currentDemo) {
435     	demoPartList.push(currentDemo);
436     }
437     
438     /*
439      * Add demo parts. (Experimental.)
440      * 
441      * This accepts an array of SeniorDads.Demos.Demo.DemoPart in "currentDemo" into the demo screen array.
442      */
443     function addDemoParts(newDemoList) {
444     	demoPartList = newDemoList.clone();
445     }
446 
447     /*
448      * Load demo, initialise and start. 
449      * 
450      * This results in the following being called in order:
451      * # preLoader (if it exists)
452      * # loader (if it exists)
453      * # init (if it exists)
454      * # start (if it exists)
455      */
456     function loadAndStart() { 
457     	preLoad( function() { load( initAndStart ); } ); 
458     }
459     
460     /*
461      * Run demo preloader, and then run the callback in "preLoaderDone" when finished.
462      */
463     function preLoad( preLoaderDone ) {
464         if (preLoaderDone === undefined) preLoaderDone = null;	// "preLoaderDone" is optional.
465     	if (preLoader != null)  		// Only run if we have an preloader function
466     		preLoader( preLoaderDone );
467     	else  							// If there's no preloader, try running the callback
468     		if (preLoaderDone != null)
469     			preLoaderDone();
470     }
471     
472     /*
473      * Run demo loader, and then run the callback in "loaderDone" when finished.
474      */
475     function load( loaderDone ) {
476         if (loaderDone === undefined) loaderDone = null;	// "loaderDone" is optional.
477     	if (loader != null)  			// Only run if we have an loader function
478     		loader( loaderDone );
479     	else  							// If there's no loader, try running the callback
480     		if (loaderDone != null)
481     			loaderDone();
482     }
483 
484     /*
485      * Initialise and start demo. 
486      * 
487      * This results in the following being called in order:
488      * # init (if it exists)
489      * # start (if it exists)
490      */
491     function initAndStart() {
492     	initialise(); 
493 		start();
494     }
495 
496     /*
497      * Initialise demo without starting. 
498      */
499     function initialise() {
500     	if ( init != null)		// Only run if we have an initialise function
501     		init(); 
502     }
503     
504     /*
505      * Start demo.
506      * 
507      * This resets the demo array position to the start, and pops off the 
508      * next demo screen. (ie the first one.)
509      */
510     function start() {
511     	currentDemoPartCounter = 0;			// Start of demo
512     	started = true;
513     	nextDemoPart();						// Start first demo screen
514     }
515     
516     /*
517      * Get next demo screen.
518      * 
519      * This pops off the next screen in the array. If we're out of screens,
520      * it evokes the demo abort.
521      */
522     function nextDemoPart() {
523     	if (currentDemoPartCounter < demoPartList.length ) {			// If We've still got screens,...
524 	    	currentDemoPart = demoPartList[currentDemoPartCounter++];	// ... get the next screen,...
525 	    	currentDemoPart.start();									// ... and start it.
526     	}
527     	else {
528     		abortDemo();
529     	}
530     }
531     
532     /*
533      * Abort demo.
534      * 
535      * If we're got any screens currently running, abort that as well.
536      */
537     function abortDemo() {
538     	if ( currentDemoPart != null )			// If we're got any screens currently running...
539     		currentDemoPart.abort();			// .. abort that as first.
540     	if ( abort != null)						// Only run if we have an abort function
541     		abort();
542     	ended = true;
543     	if (onEnd != null)						// Only run if we have an onEnd function
544     		onEnd();
545     }
546     
547     /**
548     * <h1>SeniorDads.Demos.Demo.DemoPart</h1>
549     * <p><strong>Senior Dads demo screen.</strong></p>
550     * <p><em>var myDemoPart = new SeniorDads.Demos.Demo.DemoPart(init, start, abort);</em></p>
551     * @constructor 
552     * @class Senior Dads demo screen.
553 	* @param {function} init Function to call before starting.
554 	* @param {function} start Function to call to start screen.
555 	* @param {function} abort Function to call if the screen needs to make a speedy exit.
556     */
557     SeniorDads.Demos.Demo.DemoPart = function(init, start, abort) {
558     	/**
559     	 * Call function before starting.
560     	 * @function
561     	 */
562         this.init = function () { 
563         	if ( init != null)				// Only call if we have a function.
564         		init(); 
565         };
566     	/**
567     	 * Call function to start screen.
568     	 * @function
569     	 */
570         this.start = function () { 
571         	if ( start != null)				// Only call if we have a function.
572         		start(); 
573         };
574     	/**
575     	 * Call function if the screen needs to make a speedy exit.
576     	 * @function
577     	 */
578         this.abort = function () { 
579         	if ( abort != null)				// Only call if we have a function.
580         		abort(); 
581         };
582         
583         this.init();
584     };
585 };

587 /**
588  * <h1>SeniorDads.Demos.Player</h1>
589  * <p><strong>"YouTube" style player for Senior Dads demos</strong></p>
590  * <p><em>var myPlayer = new SeniorDads.Demos.Player(container, name, demoClass, width, height);</em></p>
591  * @constructor 
592  * @class Senior Dads demo player
593  * @param {DOM object} container HTML element from the web page to put the player in
594  * @param {string} name Unique name for the demo element
595  * @param {object} demoClass Name of the demo class.
596  * @param {int} [width] Width of the player.
597  * @param {int} [height] Height of the player.
598  * @example
599  * 	<html><head><script language="javascript">
600  *  // Load the 'Def Demo into the player. (See {@link SeniorDads.Demos.DefDemo})
601  *  var player;

603  *  	player = new SeniorDads.Demos.Player(document.getElementById('demo'), 'democanvas', SeniorDads.Demos.DefDemo);
604  *  }
605  *  </script></head>
606  *  <body onload="startPlayer();"><div id="demo"></div></body>
607  *  </html>
608  */
609 SeniorDads.Demos.Player = function(container, name, demoClass, width, height) {
610 	var button;				// Button element
611 	var replay = false;		// False as we're initalising
612 	var buttonOpacity = 1;	// We can fade the button in/out
613 
614 	// Create new instance of the demo, and start loading
615 	var demo = new demoClass(container, name, playerEnd);	// Set the replay function up when the demo ends. 
616 	demo.Load(readyToPlay); 								// When loaded, activate the play button.
617 	
618 	if (width === undefined) width = demoClass.width;		// If width, height and name not specified, try getting it from the demo.
619 	if (height === undefined) height = demoClass.height;
620 	if (width === undefined) width = 640;					// If not default is 640 x 400;
621 	if (height === undefined) height = 400;
622 	
623 	// Create the styling for the player, and add to the web page.
624     var styleSheet = document.getElementsByTagName("head")[0].appendChild(document.createElement('style'));
625     styleSheet.innerHTML = 
626     	"@keyframes fadeout { 0% { opacity: 1; } 100% { opacity: 0; } } " +
627     	"@keyframes fadein { 0%   { opacity: 0; } 100% { opacity: 1; } }" +
628     	"#" + container.id +" { width: " + width + "px; min-height: " + height + "px; border: 1px solid #888888; position: relative; box-shadow: 0px 0px 10px 2px #000000; } " +
629     	"#" + container.id +" #loadingCover { z-index: 9999; position: absolute; left: " + (width - 200) / 2 + "px; top: " + (height - 200) / 2 + "px; width: 200px; height: 200px; }" +
630     	"#" + container.id +" #loading { background-image: url(" + DEMO_ROOT + "/common/resources/demo_loading.png); width:200px; height:200px; }" +
631     	"#" + container.id +" #loaded { background-image: url(" + DEMO_ROOT + "/common/resources/demo_play.png); cursor:pointer; width:200px; height:200px; }" +
632     	"#" + container.id +" #loaded:hover { background-image: url(" + DEMO_ROOT + "/common/resources/demo_play_hover.png); }" +
633     	"#" + container.id +" #replay { background-image: url(" + DEMO_ROOT + "/common/resources/demo_replay.png); cursor:pointer; width:200px; height:200px; }" +
634     	"#" + container.id +" #replay:hover { background-image: url(" + DEMO_ROOT + "/common/resources/demo_replay_hover.png); }" +
635     	"#" + container.id +" #Fullscreen { color: #ffffff; cursor: pointer; z-index: 9998; position: absolute; right: 0px; top: 0px; -webkit-animation: fadeout 3s forwards; -moz-animation: fadeout 3s forwards; -o-animation: fadeout 3s forwards; animation: fadeout 3s forwards; }" +
636     	"#" + container.id +" #Fullscreen:hover { -webkit-animation: fadein 0.5s forwards; -moz-animation: fadein 0.5s forwards; -o-animation: fadein 0.5s forwards; animation: fadein 0.5s forwards; }" +
637     	"";
638 
639     /* in order: x offset, y offset, blur size, spread size, color */
640     /* blur size and spread size are optional (they default to 0) */
641     
642     // Now add the HTML for the player within the container
643     var loadingCover = container.appendChild(document.createElement('div'));	// The wrapper for the player
644     loadingCover.id = 'loadingCover';
645     var playButton = loadingCover.appendChild(document.createElement('div'));	// The loading/play/replay button
646     playButton.id = 'loading';
647     
648 	// The top right fullscreen button.
649    	var fullscreenButton = container.appendChild(document.createElement('div'));
650    	fullscreenButton.id = 'Fullscreen';
651    	fullscreenButton.innerHTML = "[fUlLsCr33N!!1!]";
652    	// Set up the button events
653    	fullscreenButton.addEventListener('click',  fullscreen );
654    	fullscreenButton.addEventListener('touchend',  fullscreen );
655    	// Set up events for when the screen goes in/out of fullscreen (ie for toggling the fullscreen button.)
656    	document.addEventListener("fullscreenchange", fullScreenToggled);
657    	document.addEventListener("webkitfullscreenchange", fullScreenToggled);
658    	document.addEventListener("mozfullscreenchange", fullScreenToggled);
659    	document.addEventListener("MSFullscreenChange", fullScreenToggled);
660 
661    	// When the demo is loaded, activate the play button.
662 	function readyToPlay() {
663     	playButton.id = "loaded";				// This changes the styling of the "loading" button to the "play" button.
664     	playButton.addEventListener('mousedown', runDemo);		// Event to play the demo
665     	playButton.addEventListener('touchend', runDemo);
666     	window.onkeypress = runDemo;							// Can also play the demo by pressing a key
667     }
668 
669 	// When "Play" is selected, this function is run
670     function runDemo() { 
671     	playButton.style.display = "none";						// Make button disappear!
672     	playButton.removeEventListener('mousedown', runDemo);	// De-activate play button events
673     	playButton.removeEventListener('touchend', runDemo);
674     	if (replay) {											// If we're replaying, get rid of the old instance of the demo, and reload.
675         	demo = null;    	
676         	demo = new demoClass(container, name, playerEnd); 
677         	demo.LoadAndStart(); 
678     	}
679     	else 
680     		demo.Start(); 										// Otherwise, just start the demo.
681     }
682 
683     // When the fullscreen button is clicked, this is run.
684     function fullscreen() {
685     	demo.fullscreen();			// Easy, huh!
686     }
687     
688     // When the demo ends, fade on the replay button.
689     function playerEnd() {
690     	buttonOpacity = 0;						// The startiong opacity of the button (ie completely clear)
691     	replay = true;							// Specify that we're in re-play mode.
692     	playButton.id = "replay";				// This changes the styling of the "play" button to the "replay" button.
693     	playButton.style.display = "block";					// Bring back the button...
694     	playButton.style.opacity = buttonOpacity;			// ... But make it invisible.
695     	playButton.addEventListener('mousedown', runDemo);	// Bring back the "play" events.
696     	playButton.addEventListener('touchend', runDemo);
697     	setTimeout(fadeInLoop, 3000);			// Wait a couple of seconds, then start fading the button back in.
698     	window.onkeypress = runDemo;			// Can also re-play the demo by pressing a key
699     }
700     
701     // Fade in the re-play button. This works by changing the element's opacity from 0 (clear) 
702     // to 1 (opaque) in small steps every frame.
703     function fadeInLoop() {
704     	if (buttonOpacity < 1) {						// If the button hasn't been fully faded-in...
705     		buttonOpacity += 0.005;						// ... increment the opacity a little
706     		playButton.style.opacity = buttonOpacity; 	// ... and set the button's opacity
707     		requestAnimFrame(fadeInLoop);				// Wait a frame for the next part of the fade.
708     	} 
709     }
710     
711     // When we go in/out of fullscreen, this function is run.
712     // This is used so that we can toggle the "Fullscreen" button on or off.
713     // (Obviously it not much use having it when we're already in fullscreen!) 
714     function fullScreenToggled() {
715     	if (isFullScreen()) {												// If we're in fullscreen
716     		fullscreenButton.style.display = "none";						// ... make the fullscreen button disappear
717     		fullscreenButton.removeEventListener('mousedown', fullscreen);	// ... and de-activate any events associated with it.
718     		fullscreenButton.removeEventListener('touchend', fullscreen);
719     	}
720     	else {																// Other wise...
721     		fullscreenButton.style.display = "block";
722     	   	fullscreenButton.addEventListener('click',  fullscreen );		// ... make the fullscreen button re-appear
723     	   	fullscreenButton.addEventListener('touchend',  fullscreen );	// ... and re-activate any events associated with it.
724     	}
725     }
726     
727     /**
728      * Cross browser hack to detect if the browser is in fullscreen mode.
729      * @private
730      */
731     function isFullScreen() {
732     	return (
733     			document.fullscreenElement ||
734     			document.webkitFullscreenElement ||
735     			document.mozFullScreenElement ||
736     			document.msFullscreenElement
737     	);
738     }
739 
740 };
