  3 SeniorDads.Demos.DadPlazz = function(placeholder, name, callbackOnEnd) {
  4     this.blah = "dAdPlAz!";
  5     var demoWindow;
  6     this.Load = start;
  7     this.LoadAndStart = start;
  8     this.Start = start;
  9     this.Abort = abort;
 10 
 11     /* Main demo
 12      */    		
 13     function start() {
 14     	demoWindow = new SeniorDads.ScreenHandler(placeholder, name, 320, 240);
 15         var plazzScreen = demoWindow.add("dp", 2);
 16         var vblTimer = Math.floor( (Math.random() * 0xffff) + 1);
 17         var srcScreenPointer = 0;
 18         var destScreenPointer = srcScreenPointer + (320 * 4);
 19         var context = plazzScreen.context;
 20         var canvasData = context.createImageData(640, 240);
 21         var dataData = canvasData.data;
 22         plazzScreen.Show();
 23         plazzBlit();
 24         
 25         function plazzBlit() {
 26 	        for (var i = (320*101); i > 0; i--) {
 27 	        	var src = ( srcScreenPointer + i ) * 4;
 28 	        	var mult = ( i * 4 ) * 4;
 29 	        	var decMult = mult - (2 * 4);
 30 	        	copyBit(src, src + mult);
 31 	        	copyBit(src, src + decMult);
 32 	        	copyBit(src, destScreenPointer + mult);
 33 	        	copyBit(src, destScreenPointer + decMult);
 34 	        	plotBit(vblTimer,src);
 35 	        	i--;
 36 	        }
 37 	        context.putImageData(canvasData, 0, 0);
 38             context.drawImage(plazzScreen.canvas, 0, 0, 320, 480);
 39 	        vblTimer++;
 40 	        vblTimer = vblTimer & 0xffff;
 41 	        demoWindow.WaitVbl(5, plazzBlit);
 42         }
 43         
 44         function copyBit(src, dest) {
 45         	dataData[dest++] = dataData[src++];
 46         	dataData[dest++] = dataData[src++];
 47         	dataData[dest++] = dataData[src++];
 48         	dataData[dest] = dataData[src];
 49         }
 50         
 51         function plotBit(truColourVal, dest) {
 52             dataData[dest++] = ((truColourVal & 63488) >> 8);
 53             dataData[dest++] = ((truColourVal & 2016) >> 3);
 54             dataData[dest++] = ((truColourVal & 31) << 3);
 55             dataData[dest] = 0xff;
 56         }
 57         
 58     }
 59     
 60     function abort() {
 61         demoWindow.FlushWaitVbl();
 62     }
 63 
 64 };
