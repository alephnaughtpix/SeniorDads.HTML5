 58         var pauseTable = new Uint16Array([5, 1, 2, 1, 5, 1, 3, 4, 5, 6, 2, 1, 1, 1, 1, 4, 50, 1]);

 67         function crapOnScreen() {
 68             var pause;
 69             var length = pauseTable.length;
 70             var pausePointer = 0;
 71             nextCrapStep();
 72 
 73             function nextCrapStep() {
 74                 if (pausePointer < length) {
 75                     pause = pauseTable[pausePointer];
 76                     demoWindow.WaitVbl(pause, plotCrap);
 77                 } else {
 78                     end();
 79                 }
 80             }
 81 
 82             function plotCrap() {
 83                 plotPoint(truColourToWebColour(pause));
 84                 for (var j = (640 / 4 - 1); j > -1; j--) {
 85                     plotPoint(truColourToWebColour(j));
 86                     pause = pause ^ j;
 87                     pause = !pause;
 88                     plotPoint(truColourToWebColour(pause));
 89                 }
 90                 pausePointer++;
 91                 nextCrapStep();
 92             }

101         function truColourToWebColour(truColourVal) {
102             //Value is in format: RRRRrGGGGGgBBBBb
103             //& %1111100000000000 >> 12 = %RRRR -> 0-0xf
104             //& %0000011111100000 >> 7  = %GGGG -> 0-0xf
105             //& %0000000000011111 >> 1  = %BBBB -> 0-0xf
106             var r = ((truColourVal & 63488) >> 12).toString(16);
107             var g = ((truColourVal & 2016) >> 7).toString(16);
108             var b = ((truColourVal & 31) >> 1).toString(16);
109             return '#' + r + r + g + g + b + b;
110         }
